4   import edu.pdx.cs410J.ParserException;

6   import java.io.BufferedReader;
7   import java.io.FileReader;
8   import java.io.IOException;
9   import java.io.Reader;
10  
11  /**
12   * A skeletal implementation of the <code>TextParser</code> class for Project 2.
13   */
14  public class TextParser implements AppointmentBookParser<AppointmentBook> {
15    private final Reader reader;
16  
17    public TextParser(Reader reader) {
18      this.reader = reader;
19    }
20  
21    @Override
22    public AppointmentBook parse() throws ParserException {
23      try (
24        BufferedReader br = new BufferedReader(this.reader)
25      ) {
26  
27        String owner = br.readLine();
28  
29        if (owner == null) {
30          throw new ParserException("Missing owner");
31        }
32  
33        return new AppointmentBook(owner);
34  
35      } catch (IOException e) {
36        throw new ParserException("While parsing appointment book text", e);
37      }
38    }
39  }
