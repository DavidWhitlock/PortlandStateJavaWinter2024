15  /**
16   * Unit tests for the Student class.  In addition to the JUnit annotations,
17   * they also make use of the <a href="http://hamcrest.org/JavaHamcrest/">hamcrest</a>
18   * matchers for more readable assertion statements.

20  public class StudentTest
21  {
22  
23    private Student dave;
24  
25    @BeforeEach
26    void createDaveStudent() throws InvalidGpaException, InvalidGenderException {
27      ArrayList<String> classes = new ArrayList<>();
28      classes.add("Algorithms");
29      classes.add("Operating Systems");
30      classes.add("Java");
31      this.dave = new Student("Dave", classes, 3.64, "male");
32    }
33  
34    @Test
35    void studentNamedPatIsNamedPat() throws InvalidGpaException, InvalidGenderException {
36      // GIVEN: I've created a Student with a name of "Pat"
37      String name = "Pat";
38      var pat = new Student(name, new ArrayList<>(), 0.0, "Doesn't matter");
39  
40      // WHEN: I get the Student's name
41      // THEN: The name is "Pat"
42      assertThat(pat.getName(), equalTo(name));
43    }
44  
45    @Test
46    void studentWithGpaGreaterThan4ThrowsInvalidGpaException() throws InvalidGenderException {
47      // GIVEN: A student named "Sharon" who is
48      //        taking "Java" and "Front End"
49      //        has a GPA of 5.00
50      //        has a "female" gender
51      ArrayList<String> classes = new ArrayList<>();
52      classes.add("Java");
53      classes.add("Front End");
54  
55      // WHEN: That student is created
56      // THEN: A InvalidGpaException is thrown
57      try {
58        new Student("Sharon", classes, 5.0, "female");
59        fail("Should have thrown an InvalidGpaException");
60  
61      } catch (InvalidGpaException ex) {
62        // We expect this exception
63      }
64    }
65
